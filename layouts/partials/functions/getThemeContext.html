{{/* Get Theme Context Function =========================== Returns both the current section and resolved theme name. This function handles: 1. Section resolution (with overridable sections logic) 2. Theme resolution from section params -> site params -> default Returns: dict with "Section" and "Theme" keys Usage: {{ $ctx := partial "functions/getThemeContext" . }} {{ $ctx.Section }} - Section object {{ $ctx.Theme }} - Theme name string (e.g., "horizon2020") */}} {{/* Resolve current section */}} {{- $overridableSections := slice "authors" "catalogue" "genres" -}} {{- $currSection := .Section | default (urlize .Site.Params.currentIssue) -}} {{- if in $overridableSections $currSection -}} {{- $currSection = default (urlize .Site.Params.currentIssue) -}} {{- end -}} {{- $currSection := .Site.GetPage "section" (print $currSection "/__index.md") -}} {{/* Resolve theme from section -> site -> default */}} {{- $theme := "nebula2026" -}} {{- if $currSection -}} {{- $theme = $currSection.Params.theme | default .Site.Params.defaultTheme | default "nebula2026" -}} {{- else -}} {{- $theme = .Site.Params.defaultTheme | default "nebula2026" -}} {{- end -}} {{/* Return both section and theme */}} {{- return dict "Section" $currSection "Theme" $theme -}}
