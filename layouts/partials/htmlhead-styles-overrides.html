{{/* HTML Head Styles Overrides - Theme Dispatcher Wrapper ===================================================== This partial delegates to the theme-specific styles implementation. Note: This partial receives $currSection as context (not page context), so we need to handle it differently. */}} {{- $currSection := . -}} {{/* Build a pseudo-context for the dispatcher */}} {{- $theme := $currSection.Params.theme | default $.Site.Params.defaultTheme | default "nebula2025" -}} {{- $partialPath := printf "themes/%s/styles.html" $theme -}} {{- $fallbackPath := "themes/nebula2025/styles.html" -}} {{/* Prepare context matching what theme-dispatch provides */}} {{- $fullContext := dict "Page" $currSection "Section" $currSection "Theme" $theme "Data" dict -}} {{/* Try theme-specific partial, fallback to nebula2025 */}} {{- if templates.Exists (printf "partials/%s" $partialPath) -}} {{- partial $partialPath $fullContext -}} {{- else if templates.Exists (printf "partials/%s" $fallbackPath) -}} {{- partial $fallbackPath $fullContext -}} {{- else -}} {{- end -}}
