{{/* HTML Head Styles Overrides - Theme Dispatcher Wrapper ===================================================== This partial delegates to the theme-specific styles implementation. Receives dict with "Section" and "Page" keys from htmlhead.html. */}} {{- $currSection := .Section -}} {{- $currPage := .Page -}} {{/* Build a pseudo-context for the dispatcher */}} {{- $theme := $currSection.Params.theme | default $.Site.Params.defaultTheme | default "nebula2026" -}} {{- $partialPath := printf "themes/%s/styles.html" $theme -}} {{- $fallbackPath := "themes/nebula2026/styles.html" -}} {{/* Prepare context matching what theme-dispatch provides */}} {{- $fullContext := dict "Page" $currPage "Section" $currSection "Theme" $theme "Data" dict -}} {{/* Try theme-specific partial, fallback to nebula2026 */}} {{- if templates.Exists (printf "partials/%s" $partialPath) -}} {{- partial $partialPath $fullContext -}} {{- else if templates.Exists (printf "partials/%s" $fallbackPath) -}} {{- partial $fallbackPath $fullContext -}} {{- else -}} {{- end -}}
